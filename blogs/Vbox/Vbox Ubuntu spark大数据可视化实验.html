def toDate(inputStr):
    newStr = ""
    if len(inputStr) == 8:       #æ—¥æœŸæ ¼å¼ï¼š2020/2/2  ------>  2020-02-02
        s1 = inputStr[0:4]
        s2 = inputStr[5:6]
        s3 = inputStr[7]
        newStr = s1+"-"+"0"+s2+"-"+"0"+s3
    else:      #æ—¥æœŸæ ¼å¼ï¼š2020/1/22  ------>  2022-02-22
        s1 = inputStr[0:4]
        s2 = inputStr[5:6]
        s3 = inputStr[7:]
        newStr = s1+"-"+"0"+s2+"-"+s3
    date = datetime.strptime(newStr, "%Y-%m-%d")
    return date
